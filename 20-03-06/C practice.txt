#include<stdio.h>                                                                                                                                                                                                                                                                                                                                                                                                                   

int main()
{
	//1.1부터 100까지 홀수만 출력하기
	for (int i = 1;i < 101;i++)
	{
		if (i % 2 == 0)				//짝수 배제
			continue;
		printf("%d\n", i);
	}

	//2.1부터 100까지 3의 배수만 출력하기
	for (int j = 3;j < 101;j++)
	{
		if (j % 3 != 0)				//3의 배수가 아닌 것 배제
			continue;
		printf("%d\n", j);
	}

	//3.다음과 같은 형식으로 출력하시오.이중for문
	for (int k = 1;k < 6;k++)		//줄표시 for문
	{
		for (int t = 0;t < k;t++)	//별갯수 for 문
			printf("*");

		printf("\n");
	}

	//4.다음과 같은 형식으로 출력하시오.이중for문
	for (int q = 5;q > 0;q--)		//줄표시 for문
	{
		for (int w = 0;w < q;w++)	//별갯수 for문
			printf("*");

		printf("\n");
	}

	//5.다음과 같은 형식으로 출력하시오.이중for문
	int c, d, e,f;

	/*
	scanf_s("%d", &c);			//줄 수를 입력받을 경우
	f = c + 1;
	*/
	f = 6;

	for (c=5 ;c > 0;c--)			//줄 표시 for문
	{
		for (d = c-1;d > 0; d--)		//앞쪽 공백 for문
		{
			printf(" ");
		}

		for(e=1;e<=(2*(f-c))-1;e++)		//별갯수 for문
		{
			printf("*");
		}
		printf("\n");

	}


	//6.구구단을 가로방향으로 출력하시오.
	int a, b;

	for (a = 1;a < 10;a++)
	{
		for (b = 1;b < 10;b++)
			printf("%2dX%2d=%2d\t", a, b, a * b);

		printf("\n");
	}
}
