입력받아서 평균을 출력하시오.

2. 정수 a의 값을 짝수/홀수 판별하기

3. 정수 a의 값을 키보드로 입력받아서 
짝수/홀수 판별하기



#include<stdioh.h>

int main()
{
	//1.국어, 영어, 수학 점수를 입력받아서 평균을 출력하시오.
	int	k = 0, e = 0, m = 0, avg = 0;

	printf("국어 점수를 입력하시오 :");	//국어점수 입력
	scanf_s(" %d", &k);

	printf("\n영어 점수를 입력하시오 :");	//영어점수 입력
	scanf_s(" %d", &e);

	printf("\n수학 점수를 입력하시오 :");	//수학점수 입력
	scanf_s(" %d", &m);

	printf("\n국어 점수 : %d\n영어 점수 : %d\n수학 점수 : %d\n점수들의 평균은 %d입니다.\n",k,e,m,avg = (k + e + m) / 3);	//평균 계산&출력


	//2.정수 a의 값을 짝수/홀수 판별하기
	int a;

	for (a = 0;a < 100;a++)							//정수 발생
	{
		if (a % 2 == 0)							//조건이 짝수
			printf("%d는 짝수입니다.\n", a);
		else if (a % 2 == 1)							//조건이 홀수
			printf("%d는 홀수입니다.\n", a);
		else                                            //오류
			printf("잘못 입력되었습니다.\n");
	}
	

	//3.정수 a의 값을 키보드로 입력받아서 짝수/홀수 판별하기
	int a = 0;

	printf("정수를 입력하시오 : ");						//정수 입력
	scanf_s(" %d", &a);

	if (a % 2 == 0)							//조건이 짝수
		printf("\n입력 받은 %d는 짝수입니다.\n", a);
	else if (a % 2 == 1)							//조건이 홀수
		printf("\n입력 받은 %d는 홀수입니다.\n", a);
	else                                            //오류
		printf("\n잘못 입력되었습니다.\n");

	return 0;


}
