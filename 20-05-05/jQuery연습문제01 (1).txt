        $(document).ready(function () {
            //1번 문제 웹페이지 배경색 녹색
            $("#btn1").click(function () {
                $("#screen").addClass("screen");
            });
            $("#btn2").click(function () {
                $("#screen").removeClass("screen");
            });

            //2번 테이블 태그 짝수행 배경색이 오랜지색
            $.each($("#table tr"), function (i, o) {
                if (i % 2 == 1) {
                    $(document.getElementsByTagName("tr")[i]).addClass("tr");
                }
            });

            //3번 이미지에 그림자 넣기
            $("#btn3").click(function () {
                $("#shadow").addClass("shadow");
            });
            $("#btn4").click(function () {
                $("#shadow").removeClass("shadow");
            });

            //4번 첫행 배경색 검은색,글자 흰색
            $("#btn5").click(function () {
                $("#table tr").eq(0).addClass("trline");
            });
            $("#btn6").click(function () {
                $("#table tr").eq(0).removeClass("trline");
            });

            //5번 이미지가 오른쪽으로 이동
            $("#btn7").click(function () {
                $("#shadow").addClass("move");
            });
            $("#btn8").click(function () {
                $("#shadow").removeClass("move");
            });

            //6번 아코디언 구현
            $("#acbtn1").click(function(){
                $("#panel1").css("display","block")
                $("#panel2").css("display","none")
                $("#panel3").css("display","none")
            });
            $("#acbtn2").click(function(){
                $("#panel1").css("display","none")
                $("#panel2").css("display","block")
                $("#panel3").css("display","none")
            });
            $("#acbtn3").click(function(){
                $("#panel1").css("display","none")
                $("#panel2").css("display","none")
                $("#panel3").css("display","block")
            });


        });
    </script>
